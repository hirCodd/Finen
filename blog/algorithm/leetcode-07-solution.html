# Leetcode-07-solution

# 题目描述

给出一个 32 位的有符号整数，你需要将这个整数中每位上的数字进行反转。

输入: 123 输出: 321

# 解法

class Solution {
    public int reverse(int x) {
                int flag = x > 0 ? 1 : -1; //作为标志位
        int number = x > 0 ? x : -x; //只把x当做正整数处理
        int result;

        String str = Integer.toString(number);
        char[] chars = str.toCharArray();
        int n = chars.length-1;
        int length = chars.length % 2 == 0 ? chars.length/2 : (chars.length-1)/2;
        for (int i = 0; i < length; i++) {
            char temp = chars[i];
            chars[i] = chars[n-i];
            chars[n-i] = temp;
        }
        String strReverse = new String(chars);
        strReverse = strReverse.trim();
        try {
            result = Integer.valueOf(strReverse);
        } catch (Exception e) {
            return 0;
        }
        if (flag == 1) {
            return result * flag;
        } else {
            return result * flag;
        }
    }

}
